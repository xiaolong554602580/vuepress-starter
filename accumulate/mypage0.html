# 小程序图片瀑布流

WXML

<view class="wallContainer">
  <view class="item">
    <view class="picture" wx:for="{{picture}}" wx:key="index" wx:if="{{(index%2)==0}}">
      <image src="{{item.img}}" mode="widthFix"></image>
    </view>
  </view>
  <view class="item">
    <view class="picture" wx:for="{{picture}}" wx:key="index" wx:if="{{(index%2)!=0}}">
      <image src="{{item.img}}" mode="widthFix"></image>
    </view>
  </view>
</view>
